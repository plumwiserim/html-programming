## 배열을 사용하여 폭탄 위치 설정

```html
var num = [1,0,0,0,0,0,0,0,0];

function shuffle() {
	for (var i=0; i<num.length; i++) {
		var idx1 = Math.floor(Math.random()*9);
		var idx2 = Math.floor(Math.random()*9);

		if (idx1!=idx2) {
			var temp=num[idx1];
			num[idx1]=num[idx2];
			num[idx2]=temp;
		}
	}
	console.log(num);
}
```
