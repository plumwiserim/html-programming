# 가위바위보 게임 만들기

```html
      var total = 0;
      var win = 0;
```

```html
      $(document).ready(function () {
        $("p>img").click(function () {
          total++;
          var userimg = $(this).attr("src");
          $("#img1").attr("src", userimg);

          userimg = userimg.replace("./img/", "");
          userimg = userimg.replace(".png", "");

          var user = 0;
          switch (userimg) {
            case "scissors":
              user = 1;
              break;
            case "rock":
              user = 2;
              break;
            case "paper":
              user = 3;
              break;
          }

          var com = Math.floor(Math.random() * 3) + 1;
          var comimg = "./img/";
          switch (com) {
            case 1:
              comimg = comimg + "scissorscom.png";
              break;
            case 2:
              comimg = comimg + "rockcom.png";
              break;
            case 3:
              comimg = comimg + "papercom.png";
              break;
          }

          $("#img2").attr("src", comimg);

          var str = "";
          if (user == com) {
            str = str + "비김<br>";
          } else if (
            (user == 1 && com == 3) ||
            (user == 2 && com == 1) ||
            (user == 3 && com == 2)
          ) {
            str = str + "승<br>";
            win++;
          } else {
            str = str + "패<br>";
          }
          _rate = (win / total) * 100;
          rate = _rate.toFixed(2);
          str =
            str +
            "전체횟수 : " +
            total +
            ", 이긴횟수 : " +
            win +
            ", 승률 : " +
            rate;
          $("#msg").html(str);
        });
      });
```
